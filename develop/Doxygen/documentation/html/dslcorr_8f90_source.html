2 ! SUBROUTINE DSEVAL *
3 ! Authors: Thomas R. Cameron, Nikolas I. Steckley *
4 ! Date: 8/10/2017 *
5 !************************************************************************
6 ! Compute Laguree Correction term of real scalar polynomial at real *
7 ! number while ''dividing out'' previously computed roots. *
8 !************************************************************************
9 ! Input Variables: *
10 ! p: REAL(re8), array of dimension (deg+1), *
11 ! contains polynomial coefficients from constant to leading. *
12 ! alpha: REAL(re8), array of dimension (d+1), *
13 ! contains modulie of polynomial coefficients. *
14 ! tol: REAL(re8), convergence tolerance. *
15 ! conv: LOGICAL, convergence parameter. *
16 ! deg: INTEGER(in4), degree of the polynomial *
17 ! ind: INTEGER(in4), index of eigenvalue being updated *
18 ! *
19 ! Output Variables: *
20 ! er: REAL(re8), array of dimension deg, *
21 ! contains the real part of the eigenvalue approximations. *
22 ! ei: REAL(re8), array of dimension deg, *
23 ! contains the imaginary part of the eigenvalue approximations. *
24 ! berr: REAL(re8), backward error in current eigenvalue approximation.*
25 ! *
26 ! MEMORY: O(deg), FLOPS: O(deg) *
27 !************************************************************************
28 SUBROUTINE dslcorr(p, alpha, tol, conv, deg, ind, er, ei, berr)
29 USE util
30 IMPLICIT NONE
31 !scalar arguments
32 LOGICAL, INTENT(INOUT) :: conv
33 INTEGER(KIND=in4), INTENT(IN) :: deg, ind
34 REAL(KIND=re8), INTENT(IN) :: tol
35 REAL(KIND=re8), INTENT(INOUT) :: berr
36 !array arguments
37 REAL(KIND=re8), INTENT(IN) :: p(*), alpha(*)
38 REAL(KIND=re8), INTENT(INOUT) :: er(*), ei(*)
39 !local scalars
40 INTEGER(KIND=in4) :: k
41 REAL(KIND=re8) :: a, b, c, t
42 COMPLEX(KIND=re8) :: x1, x2, y1, y2
43 !intrinsic procedures
44 INTRINSIC :: dabs, dble, dcmplx, dimag, min, zabs, zsqrt
45 !external subroutines
46 EXTERNAL :: dseval, drevseval
47 
48 !initiate variables
49 x1=zero; x2=zero
50 DO k=1,ind-1
51  y1=dcmplx(er(ind)-er(k),-ei(k))**(-1)
52  x1=x1+y1
53  x2=x2+y1**2
54 ENDDO
55 t=er(ind)
56 !split into 2 cases
57 IF(dabs(t)>1) THEN
58  !compute a=revp, berr
59  t=t**(-1)
60  CALL drevseval(p, t, deg, 0, a)
61  CALL drevseval(alpha,dabs(t),deg,0,berr)
62  berr=min(dabs(a)*berr**(-1), dabs(a))
63  IF(berr<eps) THEN
64  ei(ind)=zero
65  conv=.true.
66  RETURN
67  ENDIF
68  !compute b=revp', c=revp''
69  CALL drevseval(p, t, deg, 1, b)
70  CALL drevseval(p, t, deg, 2, c)
71  !compute y1=p'/p and y2=(p'/p)'
72  y1=t*(deg-t*(b*a**(-1)))
73  y2=t**2*(deg-2*t*(b*a**(-1))+t**2*((b*a**(-1))**2-(c*a**(-1))))
74 ELSE
75  !compute a=p, berr
76  CALL dseval(p, t, deg, 0, a)
77  CALL dseval(alpha, dabs(t), deg, 0, berr)
78  berr=min(dabs(a)*berr**(-1), dabs(a))
79  IF(berr<eps) THEN
80  ei(ind)=zero
81  conv=.true.
82  RETURN
83  ENDIF
84  !compute b=p', c=p''
85  CALL dseval(p, t, deg, 1, b)
86  CALL dseval(p, t, deg, 2, c)
87  !compute y1=p'/p and y2=(p'/p)'
88  y1=b*a**(-1)
89  y2=y1**2-(c*a**(-1))
90 ENDIF
91 !remove previously found roots
92 x1=y1-x1
93 x2=y2-x2
94 k=deg-(ind-1)
95 !denominator of Laguerre correction term
96 y1=zsqrt((k-1)*(k*x2-x1**2))
97 y2=x1-y1
98 y1=x1+y1
99 IF(zabs(y1)>zabs(y2)) THEN
100  y1=k*y1**(-1)
101  IF(zabs(y1)<tol) THEN
102  ei(ind)=zero
103  conv=.true.
104  ELSE
105  er(ind)=er(ind)-dble(y1)
106  ei(ind)=-dimag(y1)
107  ENDIF
108 ELSE
109  y2=k*y2**(-1)
110  IF(zabs(y2)<tol) THEN
111  ei(ind)=zero
112  conv=.true.
113  ELSE
114  er(ind)=er(ind)-dble(y2)
115  ei(ind)=-dimag(y2)
116  ENDIF
117 ENDIF
118 RETURN
119 END SUBROUTINE dslcorr
