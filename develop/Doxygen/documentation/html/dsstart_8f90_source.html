6 ! Compute the initial esimates of the roots of a real scalar polynomial *
7 ! using the Newton Polygon method. *
8 !************************************************************************
9 ! Input Variables: *
10 ! alpha: REAL(re8) array, absolute value of the polynomial coeffs *
11 ! deg: INTEGER(in4), degree of polynomial *
12 ! *
13 ! Output Variables: *
14 ! er: REAL(re8) array, real part of the initial estimates *
15 ! ei: REAL(re8) array, imaginary part of the initial estiamtes *
16 ! *
17 ! Memory: O(deg), FLOPS: O(deg) *
18 !************************************************************************
19 SUBROUTINE dsstart(alpha, deg, er, ei)
20 USE util
21 IMPLICIT NONE
22 !scalar arguments
23 INTEGER(KIND=in4), INTENT(IN) :: deg
24 !array arguments
25 REAL(KIND=re8), INTENT(IN) :: alpha(*)
26 REAL(KIND=re8), INTENT(INOUT) :: er(*), ei(*)
27 !parameters
28 REAL(KIND=re8), PARAMETER :: pi2 = 6.2831853071795865_re8, sigma = 0.7_re8
29 !local scalars
30 INTEGER(KIND=in4) :: c, i, iold, j, nzeros
31 REAL(KIND=re8) :: ang, r, th
32 !local arrays
33 LOGICAL, DIMENSION(deg+1) :: h
34 REAL(KIND=re8), DIMENSION(deg+1):: a
35 !intrinsic procedures
36 INTRINSIC :: dcos, dexp, dlog, dsin
37 
38 !compute log(alpha)
39 DO i=1,deg+1
40  IF(alpha(i)>=eps) THEN
41  a(i)=dlog(alpha(i))
42  ELSE
43  a(i)=-one
44  ENDIF
45 ENDDO
46 !compute upper convex hull
47 CALL cnvex(deg+1,a,h)
48 !compute initial estimates
49 iold=1; c=0; th=pi2/deg
50 DO i=2,deg+1
51  IF(h(i)) THEN
52  nzeros=i-iold
53  r=dexp((a(iold)-a(i))/nzeros)
54  ang=pi2/nzeros
55  DO j=1,nzeros
56  er(c+j)=r*dcos(ang*j+th*i+sigma)
57  ei(c+j)=r*dsin(ang*j+th*i+sigma)
58  ENDDO
59  c=c+nzeros
60  iold=i
61  ENDIF
62 ENDDO
63 RETURN
64 END SUBROUTINE dsstart
