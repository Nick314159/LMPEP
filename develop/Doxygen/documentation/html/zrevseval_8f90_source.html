18 ! *

20 !************************************************************************
21 SUBROUTINE zrevseval(p, t, deg, der, a)
22 USE util
23 IMPLICIT NONE
24 !scalar arguments
25 INTEGER(KIND=in1), INTENT(IN) :: der
26 INTEGER(KIND=in4), INTENT(IN) :: deg

31 !local scalars

33 
34 IF(der==0) THEN
35  a=p(1)

37  a=t*a+p(k)
38  ENDDO
39 ELSEIF(der==1) THEN

43  ENDDO
44 ELSE
