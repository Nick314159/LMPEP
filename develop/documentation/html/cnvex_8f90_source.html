44 ! compute K such that N-2 <= 2**K < N-1
45 k = int(log(n-2.0d0)/log(2.0d0))
46 IF(2**(k+1) <= (n-2)) k = k+1
47 
48 ! For each M=1,2,4,8,...,2**K, consider the NJ pairs of consecutive
49 ! sets made up by M+1 points having the common vertex
50 ! (JC,A(JC)), where JC=M*(2*J+1)+1 and J=0,...,NJ,
51 ! NJ = MAX(0, INT((N-2-M)/(M+M))).
52 ! Compute the upper convex hull of their union by means of subroutine CMERGE
53 m = 1
54 DO i = 0, k
55  nj = max(0, int((n-2-m)/(m+m)))
56  DO j = 0, nj
57  jc = (j+j+1)*m+1
58  CALL cmerge(n, a, jc, m, h)
59  ENDDO
60  m = m+m
61 ENDDO
62 RETURN
