56 ! initiate variables
57 x1=czero; x2=czero
58 DO k=1,ind-1
59  y1=cmplx(er(ind)-er(k),-ei(k),dp)**(-1)
60  x1=x1+y1

62 ENDDO
63 DO k=ind+1,deg
64  y1=cmplx(er(ind)-er(k),-ei(k),dp)**(-1)
65  x1=x1+y1
66  x2=x2+y1**2
67 ENDDO
68 t=er(ind)
69 t2=dabs(t)
70 ! split into 2 cases
71 IF(t2>1) THEN
72  ! compute a=revp, berr
73  t=t**(-1)
74  t2=t2**(-1)
75  CALL drevseval(p, t, deg, 0, a)
76  CALL drevseval(alpha, t2, deg, 0, berr)
77  berr=dabs(a)*berr**(-1)
78  IF(berr<eps) THEN
79  ei(ind)=zero
80  check=.false.
81  RETURN
82  ENDIF
83  ! compute b=revp', c=revp''
84  CALL drevseval(p, t, deg, 1, b)
85  CALL drevseval(p, t, deg, 2, c)
86  ! compute y1=p'/p and y2=-(p'/p)'

88  c=c*a**(-1)
89  y1=t*(deg-t*b)
90  y2=t**2*(deg-2*t*b+t**2*(b**2-c))
91 ELSE

93  CALL dseval(p, t, deg, 0, a)
94  CALL dseval(alpha, t2, deg, 0, berr)

104  ! compute y1=p'/p and y2=-(p'/p)'
105  b=b*a**(-1)
106  c=c*a**(-1)

108  y2=b**2-c
109 ENDIF
110 ! remove previously found roots
111 x1=y1-x1
112 x2=y2-x2
113 ! denominator of Laguerre correction term
114 y1=zsqrt((deg-1)*(deg*x2-x1**2))
115 y2=x1-y1
116 y1=x1+y1
117 ! choose term that maximizes denominator
118 IF(zabs(y1)>zabs(y2)) THEN
119  y1=deg*y1**(-1)
120  IF(zabs(y1)<tol) THEN
121  ei(ind)=zero
122  check=.false.
123  ELSE
124  er(ind)=er(ind)-dble(y1)
125  ei(ind)=-aimag(y1)
126  ENDIF
127 ELSE

129  IF(zabs(y2)<tol) THEN
130  ei(ind)=zero
131  check=.false.
132  ELSE
133  er(ind)=er(ind)-dble(y2)
134  ei(ind)=-aimag(y2)
135  ENDIF
136 ENDIF
137 RETURN
